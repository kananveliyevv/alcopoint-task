☁ Supabase Database

  1. Supabase Dashboard → layihəniz → SQL Editor
  2. Aşağıdakı SQL-i yapışdırıb Run edin (bir dəfə kifayətdir)
  3. Səhifəni yeniləyin — status Supabase bağlı olacaq
create table if not exists public.alcopoint_db (
  id integer primary key,
  data jsonb not null default '{}'::jsonb,
  updated_at timestamptz not null default now()
);

alter table public.alcopoint_db enable row level security;

drop policy if exists "alcopoint read" on public.alcopoint_db;
drop policy if exists "alcopoint insert" on public.alcopoint_db;
drop policy if exists "alcopoint update" on public.alcopoint_db;

create policy "alcopoint read" on public.alcopoint_db for select to anon using (true);
create policy "alcopoint insert" on public.alcopoint_db for insert to anon with check (true);
create policy "alcopoint update" on public.alcopoint_db for update to anon using (true) with check (true);

insert into public.alcopoint_db (id, data) values (1, '{}'::jsonb)
on conflict (id) do nothing;

-- Vacib: policy var, amma GRANT yoxdursa yazma bloklana bilər.
grant select, insert, update on public.alcopoint_db to anon;
grant select, insert, update on public.alcopoint_db to authenticated;
